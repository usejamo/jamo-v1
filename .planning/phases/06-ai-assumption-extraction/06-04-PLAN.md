<!-- Extraction trigger pattern (from RESEARCH.md) -->
// In Step2DocumentUpload:
const extractionFiredRef = useRef(false)

useEffect(() => {
  const allComplete = documents.length > 0 && documents.every(d => d.parse_status === 'complete')
  const hasAssumptions = state.assumptions && state.assumptions.length > 0

  if (allComplete && !hasAssumptions && !extractionFiredRef.current && state.proposalId) {
    extractionFiredRef.current = true
    dispatch({ type: 'SET_EXTRACTION_STATUS', status: 'extracting' })
    supabase.functions.invoke('extract-assumptions', {
      body: { proposalId: state.proposalId }
    }).then(({ data, error }) => {
      if (error) {
        dispatch({ type: 'SET_EXTRACTION_STATUS', status: 'error' })
        return
      }
      dispatch({
        type: 'SET_ASSUMPTIONS',
        assumptions: data.assumptions.map((a: any) => ({
          id: crypto.randomUUID(),
          category: a.category,
          value: a.value,
          confidence: a.confidence >= 0.8 ? 'high' : a.confidence >= 0.5 ? 'medium' : 'low',
          source: a.source || 'document',
          status: 'pending',
        })),
        missing: data.missing || [],
      })
      dispatch({ type: 'SET_EXTRACTION_STATUS', status: 'complete' })
    }).catch(err => {
      console.error('extract-assumptions failed:', err)
      dispatch({ type: 'SET_EXTRACTION_STATUS', status: 'error' })
    })
  }
}, [documents, state.assumptions, state.proposalId])

<!-- Assumption DB persist on leaving Step 3 -->
// In wizardReducer SET_STEP case (step: 3 from step: 2):
// OR in ProposalCreationWizard as a useEffect watching step change to 3:
// Upsert approved assumptions to proposal_assumptions
const approvedAssumptions = state.assumptions.filter(a => a.status === 'approved')
if (approvedAssumptions.length > 0 && state.proposalId) {
  supabase.from('proposal_assumptions').upsert(
    approvedAssumptions.map(a => ({
      proposal_id: state.proposalId,
      org_id: orgId,
      category: a.category,
      content: a.value,   // DB column is 'content'
      confidence: a.confidence,
      status: 'approved',
      user_edited: a.source === 'user-provided',
    }))
  )
}
