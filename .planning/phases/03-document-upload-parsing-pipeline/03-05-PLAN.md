    Add extraction trigger after insert:
    ```typescript
    const { data: docData, error: docError } = await supabase
      .from('proposal_documents')
      .insert({ /* ... */ })
      .select()
      .single()

    // Trigger extraction (fire and forget — don't await)
    supabase.functions.invoke('extract-document', {
      body: { documentId: docData.id }
    }).catch(err => {
      console.error('Failed to trigger extraction:', err)
      // Update local state to show extraction failed
      setFiles(prev => prev.map(f =>
        f.file === file
          ? { ...f, status: 'error', error: 'Failed to start extraction' }
          : f
      ))
    })

    // Call onUploadComplete immediately (don't wait for extraction)
    if (onUploadComplete) {
      onUploadComplete(docData.id)
    }
    ```

    **Rationale for fire-and-forget:**
    - Extraction can take 5-30 seconds for large PDFs
    - Don't block upload UI on extraction completion
    - DocumentList polling will show status updates

    **Error handling:**
    - If Edge Function invoke fails, log error and update file status
    - User sees "Failed to start extraction" message
    - User can retry by re-uploading (or we could add a retry button in future)

    Update tests in src/components/__tests__/FileUpload.test.tsx:
    - Mock supabase.functions.invoke to return success
    - Verify invoke called with correct documentId
    - Test error case: invoke fails, status updates to 'error'

    import { createClient } from '@supabase/supabase-js'

    describe('Document Upload Pipeline Integration', () => {

          data: { path: `${mockOrgId}/${mockProposalId}/test.pdf` },
          error: null
        })

        // Mock proposal_documents insert
        vi.spyOn(supabase.from('proposal_documents'), 'insert').mockResolvedValue({
          data: {

            storage_path: `${mockOrgId}/${mockProposalId}/test.pdf`,
            mime_type: 'application/pdf',
            parse_status: 'pending'
          },
          error: null
        })

          data: { success: true, documentId: mockDocId },
          error: null

        let pollCount = 0
        vi.spyOn(supabase.from('proposal_documents'), 'select').mockImplementation(() => {
          pollCount++
          const status = pollCount === 1 ? 'pending' : pollCount === 2 ? 'extracting' : 'complete'
          return {
            eq: () => ({
              order: () => ({
                data: [{
                  id: mockDocId,
                  parse_status: status,
                  document_extracts: status === 'complete' ? [{ word_count: 100, page_count: 1 }] : []
                }],
                error: null
              })
            })
          }
        })

        // 1. Simulate FileUpload component behavior
        // Upload file to Storage

        expect(uploadData.path).toBe(`${mockOrgId}/${mockProposalId}/test.pdf`)

        // Insert metadata into proposal_documents
        const { data: docData } = await supabase
          .from('proposal_documents')
          .insert({
            org_id: mockOrgId,
            proposal_id: mockProposalId,
            name: 'test.pdf',
            storage_path: uploadData.path,
            mime_type: 'application/pdf',
            parse_status: 'pending'

          .select()
          .single()

        expect(docData.id).toBe(mockDocId)

        // Trigger extraction
        const { data: invokeData } = await supabase.functions.invoke('extract-document', {
          body: { documentId: docData.id }

          .from('proposal_documents')
          .select('*')
          .eq('proposal_id', mockProposalId)
          .order('created_at', { ascending: false })

        expect(poll2[0].parse_status).toBe('extracting')

        // Third poll: status = 'complete'

          .from('proposal_documents')
          .select('*')
