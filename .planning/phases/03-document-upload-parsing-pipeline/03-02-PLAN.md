      if (status === 'extracting') return <span className="px-2 py-1 text-xs rounded-full bg-blue-100 text-blue-600">Extracting...</span>
      if (status === 'complete') return <span className="px-2 py-1 text-xs rounded-full bg-green-100 text-green-600">Complete</span>
      if (status === 'error') return <span className="px-2 py-1 text-xs rounded-full bg-red-100 text-red-600">Failed</span>
      return null
    }

    function formatBytes(bytes: number): string {
