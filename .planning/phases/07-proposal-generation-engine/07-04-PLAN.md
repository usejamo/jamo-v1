    studyInfo: {

      indication: proposal?.indication || '',

      dueDate: proposal?.due_date || '',
      services: proposal?.services || [],
    },
    assumptions: [], // fetched by hook from proposal_assumptions
    services: proposal?.services || [],
  }
}

// Replace existing handleGenerate:
function handleGenerate() {
  const input = buildProposalInput()
  generateAll(input)
}

// Regeneration handler
function handleRegenerate(sectionKey: string) {
  const input = buildProposalInput()
  regenerateSection(sectionKey, input)
}
