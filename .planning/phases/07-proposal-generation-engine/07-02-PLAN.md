**2. Reducer: `generationReducer(state: GenerationState, action: GenerationAction): GenerationState`**
Handle all GenerationAction types:
- `SET_TONE`: update tone
- `START_GENERATION`: set isGenerating=true, currentWave=1, reset all sections to queued, completedCount=0
- `SECTION_GENERATING`: set section.status='generating', clear liveText
- `SECTION_TOKEN`: append token to section.liveText
- `SECTION_COMPLETE`: set section.status='complete', section.finalContent=content, increment completedCount
- `SECTION_ERROR`: set section.status='error', section.error=error
- `SET_ANCHOR`: update consistencyAnchor
- `WAVE_COMPLETE`: increment currentWave
- `GENERATION_COMPLETE`: set isGenerating=false, currentWave=null
- `RESET`: return initial state

Export the reducer for testing: `export function generationReducer(...)`

**3. Initial state:**
```typescript
const initialState: GenerationState = {
  isGenerating: false,
  currentWave: null,
  tone: 'formal',
  consistencyAnchor: '',
  sections: createInitialSections(),
  completedCount: 0,
  totalCount: TOTAL_SECTIONS,
}
```

**4. Helper: `readSSEStream(response: Response, onToken: (token: string) => void): Promise<void>`**
Read the SSE stream from the fetch response using `response.body.getReader()`.
Parse each line for `data: ` prefix, JSON parse, extract `content_block_delta` -> `delta.text`.
Call `onToken(text)` for each delta. Return when stream is done.
CRITICAL: Use `{ stream: true }` option on TextDecoder to handle multi-byte characters split across chunks.

**5. Helper: `fetchRagChunks(proposalId: string, sectionKey: string, therapeuticArea: string): Promise<Array<{content, doc_type, agency}>>`**
Call `supabase.functions.invoke('retrieve-context', { body: { proposalId, sectionKey, therapeuticArea, topK: 5 } })`.
Return `data.chunks` or empty array on error. Log errors but don't throw (RAG is enhancement, not blocker).

**6. Helper: `fetchAssumptions(proposalId: string): Promise<Array<{category, value, confidence}>>`**
Query `supabase.from('proposal_assumptions').select('category, content, confidence').eq('proposal_id', proposalId).eq('status', 'approved')`.
Map `content` column to `value` field. Return array.

**8. Main hook: `useProposalGeneration(proposalId: string)`**
Returns: `{ state, dispatch, generateAll, generateSection, regenerateSection }`

```typescript
export function useProposalGeneration(proposalId: string) {
  const [state, dispatch] = useReducer(generationReducer, initialState)
  const { session } = useAuth()
  const abortRef = useRef<AbortController | null>(null)

  // Supabase Realtime subscription for proposal_sections changes
  useEffect(() => {
    if (!proposalId) return
    const channel = supabase
      .channel(`proposal-sections-${proposalId}`)
      .on('postgres_changes', {
        event: '*',
        schema: 'public',
        table: 'proposal_sections',
        filter: `proposal_id=eq.${proposalId}`,
      }, (payload) => {
        const section = payload.new as any
        if (section.status === 'complete' && section.content) {
          dispatch({ type: 'SECTION_COMPLETE', sectionKey: section.section_key, content: section.content })
        }
      })
      .subscribe()
    return () => { supabase.removeChannel(channel) }
  }, [proposalId])

  // streamSection: fire one section generation, read SSE, dispatch tokens
  const streamSection = useCallback(async (
    sectionKey: string,
    proposalInput: GenerateSectionPayload['proposalInput'],
    ragChunks: GenerateSectionPayload['ragChunks'],
    anchor: string,
  ) => {
    dispatch({ type: 'SECTION_GENERATING', sectionKey })

    // Optimistic status update in DB
    await supabase.from('proposal_sections').upsert({
      proposal_id: proposalId,
      section_key: sectionKey,
      section_name: SECTION_NAMES[sectionKey],
      status: 'generating',
      org_id: session!.user!.user_metadata?.org_id, // or fetch from profile
    }, { onConflict: 'proposal_id,section_key' })

    const payload: GenerateSectionPayload = {
      proposalId,
      sectionId: sectionKey,
      proposalInput,
      ragChunks,
      consistencyAnchor: anchor,
      tone: state.tone,
    }

    const response = await fetch(
      `${import.meta.env.VITE_SUPABASE_URL}/functions/v1/generate-proposal-section`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${session!.access_token}`,
          'apikey': import.meta.env.VITE_SUPABASE_PUBLISHABLE_KEY,
        },
        body: JSON.stringify(payload),
      }
    )

    if (!response.ok || !response.body) {
      const errText = await response.text()
      dispatch({ type: 'SECTION_ERROR', sectionKey, error: errText })
      return ''
    }

    let fullText = ''
    await readSSEStream(response, (token) => {
      fullText += token
      dispatch({ type: 'SECTION_TOKEN', sectionKey, token })
    })

    // Wait up to 10s for Realtime to confirm DB write
    // If not confirmed, dispatch complete from local text
    setTimeout(() => {
      dispatch({ type: 'SECTION_COMPLETE', sectionKey, content: fullText })
    }, 10000)

    return fullText
  }, [proposalId, session, state.tone])

  // generateAll: three-wave orchestration per D-01/D-02
  const generateAll = useCallback(async (proposalInput: GenerateSectionPayload['proposalInput']) => {
    dispatch({ type: 'START_GENERATION' })

    try {
      // Fetch assumptions (for proposalInput enrichment)
      const assumptions = await fetchAssumptions(proposalId)
      const enrichedInput = { ...proposalInput, assumptions }

      // Wave 1: Understanding (serial)
      const wave1Keys = getWaveSections(1)  // ['understanding']
      const ragChunks1 = await fetchRagChunks(proposalId, wave1Keys[0], proposalInput.studyInfo.therapeuticArea)
      const wave1Text = await streamSection(wave1Keys[0], enrichedInput, ragChunks1, '')

      // Extract anchor after Wave 1
      const anchor1 = await extractAnchor(wave1Text, session!)
      dispatch({ type: 'SET_ANCHOR', anchor: anchor1 })
      dispatch({ type: 'WAVE_COMPLETE', wave: 1 })

      // Wave 2: Body sections (parallel)
      const wave2Keys = getWaveSections(2)
      const wave2Results = await Promise.all(
        wave2Keys.map(async (key) => {
          const ragChunks = await fetchRagChunks(proposalId, key, proposalInput.studyInfo.therapeuticArea)
          return streamSection(key, enrichedInput, ragChunks, anchor1)
        })
      )
      dispatch({ type: 'WAVE_COMPLETE', wave: 2 })

      // Extract full anchor after Wave 2
      const allText = [wave1Text, ...wave2Results].join('\n\n')
      const fullAnchor = await extractAnchor(allText, session!)
      dispatch({ type: 'SET_ANCHOR', anchor: fullAnchor })

      // Wave 3: Summary sections (serial)
      const wave3Keys = getWaveSections(3)
      for (const key of wave3Keys) {
        const ragChunks = await fetchRagChunks(proposalId, key, proposalInput.studyInfo.therapeuticArea)
        await streamSection(key, enrichedInput, ragChunks, fullAnchor)
      }
      dispatch({ type: 'WAVE_COMPLETE', wave: 3 })
      dispatch({ type: 'GENERATION_COMPLETE' })
    } catch (err) {
      console.error('[useProposalGeneration] generateAll error:', err)
      dispatch({ type: 'GENERATION_COMPLETE' })
    }
  }, [proposalId, session, streamSection])

  // generateSection: single section (for regeneration) — REQ-4.7
  const generateSection = useCallback(async (sectionKey: string, proposalInput: GenerateSectionPayload['proposalInput']) => {
    const ragChunks = await fetchRagChunks(proposalId, sectionKey, proposalInput.studyInfo.therapeuticArea)
    return streamSection(sectionKey, proposalInput, ragChunks, state.consistencyAnchor)
  }, [proposalId, streamSection, state.consistencyAnchor])

  // regenerateSection: reset + generate
  const regenerateSection = useCallback(async (sectionKey: string, proposalInput: GenerateSectionPayload['proposalInput']) => {
    dispatch({ type: 'SECTION_GENERATING', sectionKey })
    return generateSection(sectionKey, proposalInput)
  }, [generateSection])

  return { state, dispatch, generateAll, generateSection, regenerateSection }
}
```
