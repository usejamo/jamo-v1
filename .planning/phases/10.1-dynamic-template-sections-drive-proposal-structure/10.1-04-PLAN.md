**Step 2: Add useEffect to fetch template names for proposals that have selected_template_id:**
```typescript
useEffect(() => {
  // Collect unique template IDs from proposals
  const templateIds = [...new Set(
    proposals
      .map(p => (p as any).selected_template_id)
      .filter(Boolean)
  )]
  if (templateIds.length === 0) return

  import('../lib/supabase').then(({ supabase }) => {
    supabase
      .from('templates')
      .select('id, name')
      .in('id', templateIds)
      .then(({ data }) => {
        if (!data) return
        const names: Record<string, string> = {}
        for (const t of data) names[t.id] = t.name
        setTemplateNames(names)
      })
  })
}, [proposals])
```
