```typescript
// src/lib/chunker.ts
// Uses js-tiktoken for accurate BPE token counting (cl100k_base — matches text-embedding-3-small)
// DO NOT use character count heuristics — inaccurate for regulatory abbreviations (per RESEARCH.md)

import { get_encoding } from 'npm:js-tiktoken'  // Deno-compatible pure JS

export interface Chunk {
  content: string
  source: string        // document filename
  tokenCount: number
  sectionRef?: string   // last section heading seen before this chunk
}

const CHUNK_TARGET_TOKENS = 500
const CHUNK_MAX_TOKENS = 600
const OVERLAP_TOKENS = 100
const MIN_TOKENS = 400

// Section heading detection for regulatory docs: "4.", "4.1", "ICH E6 Section X"
const SECTION_HEADING_RE = /^(\d+\.[\d.]*)\s+[A-Z]/m

export function chunkDocument(text: string, source: string): Chunk[]
```

Implementation strategy:
1. Split text on SECTION_HEADING_RE boundaries, tracking last seen heading as sectionRef
2. For each segment: if > CHUNK_MAX_TOKENS, apply sliding window with OVERLAP_TOKENS
3. If < MIN_TOKENS, merge with next segment before chunking
4. Encode tokens with cl100k_base encoding; tokenCount = encoded.length
5. Return Chunk[] array

Note on Deno/Node compatibility: src/lib/chunker.ts is used by both the CLI (Deno) and tests (Vitest/Node). Use `npm:js-tiktoken` import for the CLI; for Vitest tests, the test can mock the token counter or use the Node-compatible import `import { get_encoding } from 'js-tiktoken'`. Keep the chunker logic decoupled from the import path — extract token counting into a small helper.
